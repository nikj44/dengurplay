const HomeScreen = () => {
  const roomlist = () => {
    return roomsdata.map((room) => {
      return (
      <View key={room.id}>
      <Text>{room.createdby}</Text>
      <Text>{room.description}</Text>
      <Text>{room.category}</Text>
      <Text>{room.characters}</Text>
      </View>
    );
    });
  };

};

  styler: {
    backgroundColor: '#3DB2FF',
  }
});
